class SystemArchitecture {
  constructor() {
    this.modules = new Map();
    this.status = 'initializing';
  }

  async loadModule(name, path) {
    try {
      const mod = await import(path);
      this.modules.set(name, mod);
      return true;
    } catch (e) {
      console.error(`Failed to load ${name}`);
      return false;
    }
  }
}
